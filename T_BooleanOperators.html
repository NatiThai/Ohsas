BooleanOperators

ตัวดำเนินการแบบ Boolean ที่ผมจะสอนมี 3 ชนิดดังนี้
and (&)
x = True
y = True
z = x and y #สำหรับand True and True จะเป็นTrue กรณีเดียว #True = 1 , False = 0
print(z) # z = True

x = True
y = False
z = x and y # x & y เขียนแบบนี้ก็ได้นะครับ
print(z) # z = False

x = False
y = True
z = x and y
print(z) # z = False

x = False
y = False
z = x and y
print(z) # z = False

or (|)
x = True
y = True
z = x or y
print(z) # z = True

x = True
y = False
z = x or y
print(z) # z = True

x = False
y = True
z = x or y
print(z) # z = True

x = False
y = False
z = x or y #สำหรับand False or False จะเป็นFalse กรณีเดียว
print(z) # z = False

not
x = True
y = not x
print(y) # y = False

x = False
y = not x
print(y) # y = True
